#include <iostream>
#include <vector>
#include <thread>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <functional>
#include <future>
#include <atomic>
#include <chrono>
#include <iomanip>

// Класс пула рабочих потоков
class WorkerPool {
private:
    std::vector<std::thread> threads;
    std::queue<std::function<void()>> jobQueue;
    std::mutex queueLock;
    std::condition_variable signal;
    std::atomic<bool> shouldStop;

public:
    WorkerPool(size_t count) : shouldStop(false) {
        for (size_t t = 0; t < count; ++t) {
            threads.emplace_back([this]() {
                while (true) {
                    std::function<void()> job;
                    {
                        std::unique_lock<std::mutex> lock(this->queueLock);
                        this->signal.wait(lock, [this]() {
                            return this->shouldStop || !this->jobQueue.empty();
                        });

                        if (this->shouldStop && this->jobQueue.empty()) {
                            return;
                        }

                        job = std::move(this->jobQueue.front());
                        this->jobQueue.pop();
                    }
                    job();
                }
            });
        }
    }

    template<class F, class... Args>
    auto submit(F&& func, Args&&... args)
        -> std::future<typename std::invoke_result_t<F, Args...>> {

        using return_type = typename std::invoke_result_t<F, Args...>;

        auto task = std::make_shared<std::packaged_task<return_type()>>(
            std::bind(std::forward<F>(func), std::forward<Args>(args)...)
        );

        std::future<return_type> result = task->get_future();

        {
            std::unique_lock<std::mutex> lock(queueLock);
            if (shouldStop) {
                throw std::runtime_error("submit on stopped WorkerPool");
            }
            jobQueue.emplace([task]() { (*task)(); });
        }

        signal.notify_one();
        return result;
    }

    ~WorkerPool() {
        shouldStop = true;
        signal.notify_all();
        for (std::thread& t : threads) {
            if (t.joinable()) t.join();
        }
    }

    size_t getThreadCount() const {
        return threads.size();
    }
};

// Вычисление числа Фибоначчи (итеративный метод)
long long fibCalc(int n) {
    if (n <= 1) return n;
    long long prev = 0, curr = 1;
    for (int step = 2; step <= n; ++step) {
        long long next = prev + curr;
        prev = curr;
        curr = next;
    }
    return curr;
}

// Утилита для рисования линии
void drawLine(char sym = '-', int width = 70) {
    std::cout << std::string(width, sym) << std::endl;
}

int main() {
    setlocale(LC_ALL, "Russian");
    system("chcp 1251 > nul");
    auto startMoment = std::chrono::steady_clock::now();

    try {
        drawLine('=', 70);
        std::cout << "          ПУЛ ПОТОКОВ: ВЫЧИСЛЕНИЕ ЧИСЕЛ ФИБОНАЧЧИ" << std::endl;
        drawLine('=', 70);

        const size_t threadAmount = 4;
        std::cout << "\n[НАСТРОЙКА] Создаём пул из " << threadAmount << " рабочих потоков." << std::endl;
        WorkerPool pool(threadAmount);
        std::cout << "[НАСТРОЙКА] Пул успешно запущен." << std::endl;

        const int totalJobs = 20;
        std::vector<std::future<long long>> jobResults;
        jobResults.reserve(totalJobs);

        std::cout << "\n--- Отправка заданий в пул ---" << std::endl;
        drawLine('-', 70);

        for (int idx = 0; idx < totalJobs; ++idx) {
            int arg = 10 + idx;
            jobResults.emplace_back(pool.submit(fibCalc, arg));
            std::cout << "  Задание " << std::setw(2) << (idx + 1) << ": Fibonacci(" << arg << ") добавлено." << std::endl;
        }

        std::cout << "\n--- Получение результатов (блокировка get) ---" << std::endl;
        drawLine('-', 70);
        std::cout << "  №   |  арг  |      Fibonacci(арг)" << std::endl;
        drawLine('-', 70);

        for (size_t pos = 0; pos < jobResults.size(); ++pos) {
            int argument = 10 + static_cast<int>(pos);
            long long value = jobResults[pos].get();
            std::cout << "  " << std::setw(2) << (pos + 1) << "   |   "
                << std::setw(2) << argument << "   |   "
                << std::setw(15) << value << std::endl;
        }

        drawLine('-', 70);
        std::cout << "[ЗАВЕРШЕНИЕ] Все " << totalJobs << " заданий выполнены." << std::endl;

        auto finishMoment = std::chrono::steady_clock::now();
        auto elapsedMs = std::chrono::duration_cast<std::chrono::milliseconds>(finishMoment - startMoment).count();
        std::cout << "\n[СТАТИСТИКА] Общее время работы: " << elapsedMs << " мс." << std::endl;

        drawLine('=', 70);
        std::cout << "Программа завершена корректно." << std::endl;
    }
    catch (const std::exception& ex) {
        std::cerr << "КРИТИЧЕСКАЯ ОШИБКА: " << ex.what() << std::endl;
        return 1;
    }

    return 0;
}
