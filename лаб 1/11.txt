#include <iostream>
#include <thread>
#include <chrono>
#include <random>

int main() {
    setlocale(LC_ALL, "Russian");
    std::cout << "Основной поток: инициализация потоков..." << std::endl;

    // Лямбда-функция, выполняющая работу потока
    auto workerTask = [](int workerId) {
        std::random_device rd;
        std::mt19937 rng(rd());
        std::uniform_int_distribution<> sleepRange(200, 300);

        for (int step = 1; step <= 5; ++step) {
            std::cout << "Рабочий " << workerId << " -> " << step << std::endl;
            std::this_thread::sleep_for(std::chrono::milliseconds(sleepRange(rng)));
        }
    };

    std::thread threadA(workerTask, 1);
    std::thread threadB(workerTask, 2);

    std::cout << "Основной поток: потоки запущены, ожидание..." << std::endl;

    for (int counter = 1; counter <= 3; ++counter) {
        std::cout << "Основной поток: сигнал " << counter << std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(150));
    }

    threadA.join();
    threadB.join();

    std::cout << "Основной поток: все задания выполнены. Завершение работы." << std::endl;
    return 0;
}
