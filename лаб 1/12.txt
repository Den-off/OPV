#include <iostream>
#include <thread>
#include <chrono>
#include <random>

int main() {
    setlocale(LC_ALL, "Russian");
    std::cout << "Основной поток: старт дочерних потоков..." << std::endl;

    // Лямбда для вывода чисел с заданной задержкой
    auto worker = [](int id, int pause) {
        for (int counter = 1; counter <= 5; ++counter) {
            std::cout << "Поток " << id << ": значение " << counter << std::endl;
            std::this_thread::sleep_for(std::chrono::milliseconds(pause));
        }
        std::cout << "Завершение работы потока " << id << std::endl;
    };

    std::thread first(worker, 1, 200);
    std::thread second(worker, 2, 800);

    std::cout << "Основной поток: оба потока созданы" << std::endl;

    first.detach();
    std::cout << "Основной поток: первый поток отделён (detach)" << std::endl;

    for (int step = 1; step <= 5; ++step) {
        std::cout << "Действие основного потока " << step << std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
    }

    std::cout << "Основной поток: ожидание второго потока (join)" << std::endl;
    second.join();

    std::cout << "Основной поток: пауза 3 секунды" << std::endl;
    std::this_thread::sleep_for(std::chrono::seconds(3));

    std::cout << "Основной поток: завершение программы" << std::endl;
    return 0;
}
